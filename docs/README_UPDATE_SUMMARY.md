# 从"计算机行业"切换到"医疗行业"仅需 3 步：

# 1. 修改行业前缀
sed -i '' 's/计算机行业/医疗行业/g' config/base_config.json

# 2. 替换术语库
cat > config/base_dict.json <<EOF
{
  "terms": ["病历", "诊断", "处方", "会诊", "CT", "核磁共振", ...]
}
EOF

# 3. 完成！无需改代码
python cli.py --info  # 立即生效
